# Audio

Audio() 构造方法和 document.createElement('audio') 都可以创建并返回一个 HTMLAudioElement 对象，该对象既可以被附加到页面上供用户交互使用，也可以游离在页面以外管理和播放音频。

检查可播放状态的方式：

  • 检查 readyState 属性
    • HTMLMediaElement.HAVE_FUTURE_DATA : 至少可播放一小段时间
    • HTMLMediaElement.HAVE_ENOUGH_DATA : 可以完整播放
  • 监听事件
    • canplay : 可以开始播放，可能中断
    • canplaythrough : 可以完整播放

推荐监听事件：

myAudioElement.addEventListener('canplaythrough', event => {
    /* the audio is now playable; play it if permissions allow */
    myAudioElement.play()
})

注意

即使所有引用都被清理，一个正在播放中的 HTMLAudioElement 也不会被垃圾回收，必需等到该音频播放完毕或暂停。

# Web Audio API

graph LR; 　　id1[创建AudioContext]-->id2[创建AnalyserNode]; 　　id2-->id3[设置SourceNode]; 　　id3-->id4[播放音频]; 　　id4-->id5[获取frequencyData]; 　　id5-->id6[绘制图形]; 　　id6-->id5; 
// 创建 AudioContext
const AudioContext = window.AudioContext || window.webkitAudioContext;
const ctx = new AudioContext();
// 创建分析器
const analyser = ctx.createAnalyser();
analyser.fftSize = 512;
// 获取<audio>节点，通过该节点创建音频源
const audio = document.getElementById('audio');
const source = ctx.createMediaElementSource(audio);
// 将音频源关联到分析器
source.connect(analyser);
// 将分析器关联到输出设备（耳机、扬声器）
analyser.connect(ctx.destination);

fft : 快速傅里叶变换看不懂，了解一下就得

fftSize : fftSize - MDN 值必须是 2 的幂次方，数值越大 frequency domain 数据越精细（决定 frequencyData 的长度，具体为 fftSize 的一半）

// 播放音频，获取频率数据
audio.play();
const bufferLength = analyser.frequencyBinCount;
const dataArray = new Uint8Array(bufferLength);
analyser.getByteFrequencyData(dataArray);

# Canvas

function draw() {
    requestAnimationFrame(draw)
    // 更新频率数据
    analyser.getByteFrequencyData(dataArray)
    canvasCtx.clearRect(0, 0, width, height)
    canvasCtx.beginPath()
    var x = 0
    // 人耳一般17000-20000hz，FrequencyData数据最后一位表示24000hz
    // 不想画直线需要舍弃一部分尾部数据
    let count = bufferLength * 0.68
    let stepWidth = width / count
    for(var i = 0; i < count; i++) {
        var v = dataArray[i] / 256;
        var y = height - v * height;
        if(i === 0) {
            canvasCtx.moveTo(x, y);
        } else {
            canvasCtx.lineTo(x, y);
        }
        x += stepWidth
    }
    canvasCtx.strokeStyle = gradient
    canvasCtx.stroke()
    canvasCtx.closePath()
}
